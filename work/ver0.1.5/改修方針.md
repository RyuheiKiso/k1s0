# k1s0 v0.1.5 改修方針

## 概要

フロントエンドフレームワーク（React/Flutter）に Real-time 通信パッケージを追加する。

- **@k1s0/realtime**: React 向け WebSocket/SSE クライアント
- **k1s0_realtime**: Flutter 向け WebSocket/SSE クライアント

## 機能目的

- リアルタイム通信（チャット、通知、ダッシュボード更新）の実装効率化
- 接続管理（自動再接続、オフライン対応）の標準化
- React/Flutter 間で一貫した API の提供

## 技術選定

### React

| 機能 | 使用技術 | 理由 |
|------|---------|------|
| WebSocket | ネイティブ WebSocket API | 標準 API、追加依存なし |
| SSE | ネイティブ EventSource API | 標準 API、自動再接続内蔵 |
| 再接続 | カスタム実装 | 細かな制御が可能 |
| 状態管理 | React Context + Hooks | @k1s0/auth-client と同じパターン |

### Flutter

| 機能 | 使用パッケージ | 理由 |
|------|---------------|------|
| WebSocket | web_socket_channel | Dart チーム公式パッケージ |
| SSE | カスタム実装 | 軽量な実装で十分 |
| ネットワーク監視 | connectivity_plus | Flutter Community 推奨 |
| 永続化 | shared_preferences | 軽量な KV ストア |
| 状態管理 | Riverpod | k1s0_state との統合 |

## 機能 1: WebSocket クライアント

### 1.1 React 版 (`@k1s0/realtime`)

#### 設計方針

- **ネイティブ WebSocket API** をラップし、便利機能を追加
- **フック中心設計**: `useWebSocket` で状態と操作を提供
- **自動再接続**: Exponential backoff アルゴリズム
- **ハートビート**: 接続維持のための ping/pong

#### 型定義

```typescript
// 接続状態
type ConnectionStatus = 'connecting' | 'connected' | 'disconnecting' | 'disconnected';

// 再接続設定
interface ReconnectConfig {
  enabled: boolean;
  maxAttempts: number;        // 最大再接続回数（0 = 無限）
  backoff: 'linear' | 'exponential';
  initialDelay: number;       // 初回遅延（ms）
  maxDelay: number;           // 最大遅延（ms）
  jitter: boolean;            // ランダム揺らぎ追加
}

// ハートビート設定
interface HeartbeatConfig {
  enabled: boolean;
  interval: number;           // 送信間隔（ms）
  timeout: number;            // 応答タイムアウト（ms）
  message: string | (() => string);
  expectedResponse?: string | ((msg: unknown) => boolean);
}

// WebSocket フックオプション
interface UseWebSocketOptions<T = unknown> {
  url: string;
  protocols?: string | string[];

  // 再接続
  reconnect?: Partial<ReconnectConfig>;

  // ハートビート
  heartbeat?: Partial<HeartbeatConfig>;

  // 認証
  getAuthToken?: () => string | Promise<string>;

  // シリアライズ
  serialize?: (data: unknown) => string;
  deserialize?: (data: string) => T;

  // イベントハンドラ
  onOpen?: (event: Event) => void;
  onClose?: (event: CloseEvent) => void;
  onError?: (event: Event) => void;
  onMessage?: (data: T) => void;
  onReconnecting?: (attempt: number) => void;
  onReconnected?: () => void;

  // 自動接続
  autoConnect?: boolean;      // デフォルト: true
}

// WebSocket フック戻り値
interface UseWebSocketReturn<T = unknown> {
  // 状態
  status: ConnectionStatus;
  lastMessage: T | null;
  error: Error | null;
  reconnectAttempt: number;

  // 操作
  connect: () => void;
  disconnect: (code?: number, reason?: string) => void;
  sendMessage: (data: unknown) => void;
  sendJson: (data: object) => void;

  // WebSocket インスタンス
  getSocket: () => WebSocket | null;
}
```

#### 使用例

```tsx
import { useWebSocket } from '@k1s0/realtime';

interface ChatMessage {
  id: string;
  userId: string;
  content: string;
  timestamp: string;
}

function ChatRoom({ roomId }: { roomId: string }) {
  const [messages, setMessages] = useState<ChatMessage[]>([]);

  const {
    status,
    sendMessage,
    reconnectAttempt,
  } = useWebSocket<ChatMessage>({
    url: `wss://api.example.com/ws/chat/${roomId}`,

    reconnect: {
      enabled: true,
      maxAttempts: 10,
      backoff: 'exponential',
      initialDelay: 1000,
      maxDelay: 30000,
    },

    heartbeat: {
      enabled: true,
      interval: 30000,
      timeout: 5000,
      message: JSON.stringify({ type: 'ping' }),
    },

    onMessage: (message) => {
      setMessages((prev) => [...prev, message]);
    },

    onReconnecting: (attempt) => {
      toast.info(`再接続中... (${attempt}回目)`);
    },
  });

  const handleSend = (content: string) => {
    sendMessage({
      type: 'chat.message',
      payload: { content },
    });
  };

  return (
    <Box>
      <ConnectionBadge status={status} attempt={reconnectAttempt} />
      <MessageList messages={messages} />
      <MessageInput onSend={handleSend} disabled={status !== 'connected'} />
    </Box>
  );
}
```

### 1.2 Flutter 版 (`k1s0_realtime`)

#### コンポーネント設計

```dart
/// 接続状態
enum ConnectionStatus {
  connecting,
  connected,
  disconnecting,
  disconnected,
}

/// 再接続設定
class ReconnectConfig {
  final bool enabled;
  final int? maxAttempts;     // null = 無限
  final BackoffType backoffType;
  final Duration initialDelay;
  final Duration maxDelay;
  final bool jitter;

  const ReconnectConfig({
    this.enabled = true,
    this.maxAttempts = 10,
    this.backoffType = BackoffType.exponential,
    this.initialDelay = const Duration(seconds: 1),
    this.maxDelay = const Duration(seconds: 30),
    this.jitter = true,
  });
}

enum BackoffType { linear, exponential }

/// ハートビート設定
class HeartbeatConfig {
  final bool enabled;
  final Duration interval;
  final Duration timeout;
  final String message;
  final bool Function(dynamic)? validateResponse;

  const HeartbeatConfig({
    this.enabled = true,
    this.interval = const Duration(seconds: 30),
    this.timeout = const Duration(seconds: 5),
    this.message = '{"type":"ping"}',
    this.validateResponse,
  });
}

/// K1s0 WebSocket クライアント
class K1s0WebSocket {
  final String url;
  final ReconnectConfig reconnectConfig;
  final HeartbeatConfig heartbeatConfig;
  final String? Function()? getAuthToken;
  final dynamic Function(String)? deserialize;
  final String Function(dynamic)? serialize;

  K1s0WebSocket({
    required this.url,
    this.reconnectConfig = const ReconnectConfig(),
    this.heartbeatConfig = const HeartbeatConfig(),
    this.getAuthToken,
    this.deserialize,
    this.serialize,
  });

  // ストリーム
  Stream<ConnectionStatus> get status => _statusController.stream;
  Stream<T> messages<T>() => _messageController.stream.cast<T>();

  // 操作
  Future<void> connect();
  void disconnect({int? code, String? reason});
  void send(dynamic data);
  void sendJson(Map<String, dynamic> data);
}
```

```dart
import 'package:k1s0_realtime/k1s0_realtime.dart';

// Riverpod Provider
final chatSocketProvider = Provider.family<K1s0WebSocket, String>((ref, roomId) {
  final socket = K1s0WebSocket(
    url: 'wss://api.example.com/ws/chat/$roomId',
    reconnectConfig: ReconnectConfig(
      enabled: true,
      maxAttempts: 10,
      backoffType: BackoffType.exponential,
    ),
    heartbeatConfig: HeartbeatConfig(

      interval: Duration(seconds: 30),

  return socket;

// メッセージストリーム Provider
final chatMessagesProvider = StreamProvider.family<ChatMessage, String>((ref, roomId) {
  final socket = ref.watch(chatSocketProvider(roomId));
  socket.connect();
  return socket.messages<Map<String, dynamic>>()
      .map((json) => ChatMessage.fromJson(json));
});

// 接続状態 Provider
final chatConnectionProvider = StreamProvider.family<ConnectionStatus, String>((ref, roomId) {
  final socket = ref.watch(chatSocketProvider(roomId));
  return socket.status;
});

// UI
class ChatPage extends ConsumerWidget {
  final String roomId;

  const ChatPage({required this.roomId});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final messagesAsync = ref.watch(chatMessagesProvider(roomId));
    final connectionAsync = ref.watch(chatConnectionProvider(roomId));

    return Scaffold(
      appBar: AppBar(
        title: Text('Chat'),
        actions: [
          connectionAsync.when(
            data: (status) => ConnectionIndicator(status: status),
            loading: () => CircularProgressIndicator(),
            error: (_, __) => Icon(Icons.error),
          ),
        ],
      ),
      body: messagesAsync.when(
        data: (message) => MessageList(messages: messages),
        loading: () => Center(child: CircularProgressIndicator()),
        error: (err, _) => ErrorWidget(error: err),
      ),
    );
  }
}
```

## 機能 2: Server-Sent Events (SSE) クライアント

### 2.1 React 版

#### 設計方針

- **EventSource API** をラップ
- ブラウザの自動再接続機能を活用
- イベントタイプ別のハンドラ登録

#### 型定義

```typescript
interface UseSSEOptions<T = unknown> {
  url: string;

  // 認証
  withCredentials?: boolean;

  // イベントハンドラ
  eventHandlers?: Record<string, (data: T) => void>;
  onMessage?: (data: T) => void;
  onError?: (event: Event) => void;

  // 再接続（EventSource 標準機能を拡張）
  reconnect?: {
    enabled: boolean;
    maxAttempts?: number;
    interval?: number;
  };

  // パーサー
  deserialize?: (data: string) => T;

  // 自動接続
  autoConnect?: boolean;
}

interface UseSSEReturn<T = unknown> {
  status: ConnectionStatus;
  lastEvent: { type: string; data: T } | null;
  error: Error | null;

  connect: () => void;
  disconnect: () => void;
}
```

#### 使用例

```tsx
import { useSSE } from '@k1s0/realtime';

interface DashboardUpdate {
  type: 'metric' | 'alert' | 'status';
  payload: unknown;
}

function Dashboard() {
  const [metrics, setMetrics] = useState<Metric[]>([]);
  const [alerts, setAlerts] = useState<Alert[]>([]);

  const { status } = useSSE<DashboardUpdate>({
    url: '/api/dashboard/stream',
    withCredentials: true,

    eventHandlers: {
      metric: (data) => {
        setMetrics((prev) => updateMetric(prev, data.payload));
      },
      alert: (data) => {
        setAlerts((prev) => [...prev, data.payload]);
        showNotification(data.payload);
      },
      status: (data) => {
        updateSystemStatus(data.payload);
      },
    },

    onError: (error) => {
      console.error('Dashboard stream error:', error);
    },
  });

  return (
    <Box>
      <StatusIndicator connected={status === 'connected'} />
      <MetricGrid metrics={metrics} />
      <AlertList alerts={alerts} />
    </Box>
  );
}
```

```dart
  final String eventType;
  final String data;
  final int? retry;

    required this.eventType,
    required this.data,
```

```dart
import 'package:k1s0_realtime/k1s0_realtime.dart';

// Notification ストリーム
final notificationProvider = StreamProvider<Notification>((ref) {
  final sse = K1s0SSE(
    url: 'https://api.example.com/notifications/stream',
    headers: {
      'Authorization': 'Bearer ${ref.read(authProvider).token}',
    },
  );

  sse.connect();
  ref.onDispose(() => sse.disconnect());

  return sse.events
      .where((e) => e.eventType == 'notification')
      .map((e) => Notification.fromJson(jsonDecode(e.data)));
});

// Dashboard ストリーム（複数イベントタイプ）
final dashboardStreamProvider = StreamProvider<DashboardEvent>((ref) {
  final sse = K1s0SSE(url: '/api/dashboard/stream');
  sse.connect();
  ref.onDispose(() => sse.disconnect());

  return sse.events.map((e) {
    switch (e.eventType) {
      case 'metric':
        return DashboardEvent.metric(Metric.fromJson(jsonDecode(e.data)));
      case 'alert':
        return DashboardEvent.alert(Alert.fromJson(jsonDecode(e.data)));
      default:
        return DashboardEvent.unknown(e);
    }
  });
});
```

## 機能 3: Realtime Provider（グローバル接続管理）

### 3.1 React 版

#### 設計方針

- **Context Provider** でグローバル設定と状態を管理
- 複数の WebSocket/SSE 接続を一元管理
- オフラインキューの共有

#### 型定義

```typescript
interface RealtimeConfig {
  // ネットワーク監視
  networkMonitor?: {
    enabled: boolean;
    onOnline?: () => void;
    onOffline?: () => void;
  };

  // オフラインキュー
  offlineQueue?: {
    enabled: boolean;
    maxSize: number;
    persistToStorage: boolean;
    storageKey?: string;
  };

  // デフォルト設定
  defaultReconnect?: Partial<ReconnectConfig>;
  defaultHeartbeat?: Partial<HeartbeatConfig>;
}

interface RealtimeContextValue {
  // ネットワーク状態
  isOnline: boolean;

  // 接続管理
  connections: Map<string, ConnectionInfo>;
  registerConnection: (id: string, info: ConnectionInfo) => void;
  unregisterConnection: (id: string) => void;
```

#### 使用例

```tsx
import { RealtimeProvider, useConnectionStatus, useOfflineQueue } from '@k1s0/realtime';

// App root
function App() {
  return (
    <RealtimeProvider
      config={{
        networkMonitor: {
          enabled: true,
          onOffline: () => toast.warning('オフラインになりました'),
          onOnline: () => toast.success('オンラインに復帰しました'),
        },
        offlineQueue: {
          enabled: true,
          maxSize: 50,
          persistToStorage: true,
        },
        defaultReconnect: {
          maxAttempts: 10,
          backoff: 'exponential',
        },
      }}
    >
      <Router />
    </RealtimeProvider>
  );
}

// グローバル接続状態インジケーター
function GlobalConnectionStatus() {
  const { isOnline, connections } = useConnectionStatus();

  const allConnected = Array.from(connections.values())
    .every((c) => c.status === 'connected');

  if (!isOnline) {
    return <Chip icon={<CloudOff />} label="オフライン" color="error" />;
  }

  if (!allConnected) {
    return <Chip icon={<Sync />} label="再接続中" color="warning" />;
  }

  return <Chip icon={<Cloud />} label="接続中" color="success" />;
}

// オフラインキュー対応のメッセージ送信
function useSendWithQueue(connectionId: string) {
  const { status, sendMessage } = useWebSocket(...);
  const { queue, flush } = useOfflineQueue();

  const send = useCallback((data: unknown) => {
    if (status === 'connected') {
      sendMessage(data);
    } else {
      queue(connectionId, data);
    }
  }, [status, sendMessage, queue, connectionId]);

  // 再接続時にキューをフラッシュ
  useEffect(() => {
    if (status === 'connected') {
      flush(connectionId, sendMessage);
    }
  }, [status, connectionId, flush, sendMessage]);

  return send;
}
```

```dart
final realtimeManagerProvider = Provider<K1s0RealtimeManager>((ref) {
  return K1s0RealtimeManager(

        enabled: true,

// ネットワーク状態 Provider
final isOnlineProvider = StreamProvider<bool>((ref) {
  return ref.watch(realtimeManagerProvider).isOnline;
});

// グローバル接続状態 Provider
final connectionsProvider = StreamProvider<Map<String, ConnectionInfo>>((ref) {
  return ref.watch(realtimeManagerProvider).connections;
});

// オフライン送信フック
extension RealtimeManagerExtension on WidgetRef {
  void sendWithQueue<T>(String connectionId, T data, void Function(T) send) {
    final manager = read(realtimeManagerProvider);
    final isOnline = read(isOnlineProvider).valueOrNull ?? false;
    final connections = read(connectionsProvider).valueOrNull ?? {};

    final connection = connections[connectionId];
    if (isOnline && connection?.status == ConnectionStatus.connected) {
      send(data);
    } else {
      manager.queue(connectionId, data);
    }
  }
}
```
